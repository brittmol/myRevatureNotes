    new project:
        - git init
        - git add .
        - git commit -m "message"
        - create new repo on github
        - git branch -M main
        - git remote add origin https://github.com/.....
        - git push -u origin main

    clone project:
        - git clone https://github.com/......

    pull other edits from repo
        - git pull

    commits:
        - git add .
        - git commit -m "message"
        - git push

    logs:
        - git log
